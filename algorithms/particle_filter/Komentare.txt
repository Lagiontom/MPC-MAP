function [public_vars] = init_particle_filter(read_only_vars, public_vars)
%INIT_PARTICLE_FILTER Inicializace částic pro filtr částic
%
% ÚČEL:
%   Vytváří počáteční soubor částic pro částicový filtr.
%   Částice musí být rovnoměrně rozptýleny ve volném prostoru mapy
%   (NE v stěnách - to by byly neplatné pozice).
%   Nejistota v iniciální poloze způsobuje, že filtr vychází
%   z "globální neznalosti" (robot neví, kde je).
%
% ALGORITMUS (Rejection Sampling - Odmítnutí neplatných pozic):
%   1. Extrahuj hranice mapy (min/max x, y koordináty)
%   2. Opakuj N-krát (dokud nemáš N částic):
%      - Vygeneruj náhodnou pozici (x, y) v obdélníku [x_min, x_max] x [y_min, y_max]
%      - Vygeneruj náhodný úhel θ v [-π, π]
%      - Ověř, zda je pozice ve VOLNÉM PROSTORU (není v nebo příliš blízko stěně)
%      - Pokud ANO: přidej particii do souboru
%      - Pokud NE: zahoď a zkus znovu
%
% KLÍČOVÉ PARAMETRY:
%   num_particles: Počet částic (vyšší = přesnější ale pomalejší)
%                  Typicky 100-1000 částic
%                  Kompromis: 500 částic = dobrá přesnost za rozumný čas
%   
%   Mapa: Musí obsahovat seznam stěn (walls) pro ověření volného prostoru
%
% INICIALIZAČNÍ STRATEGIE:
%   - Rovnoměrná distribuce: Každá poloha ve volném prostoru má stejnou pravděpodobnost
%   - Důvod: Nehádáme nic o skutečné poloze robota
%   - Filtr POTÉ converge během prvních měření (korekční krok)
%
% VISUAL (ze str. 10-12 PDF):
%   Iterace 0 (inicializace):
%      p(x) - graf "plochý" - částice všude rovnoměrně
%      □ □ □ □ □ □ □ □ □ □ (částice rozprostřeny)
%   
%   Po prvním měření (dveře):
%      p(x) - graf "dvě vrcholky" - nejistota, ale nějaký signál
%      □     □□□□     □ □    (shluk kolem dveří a korup)
%
% VSTUP:
%   read_only_vars - Struktura s mapou a konfigurací:
%     .map.limits   - [1, 4] vektor: [x_min, y_min, x_max, y_max]
%     .map.walls    - N×4 matice stěn: [x1, y1, x2, y2]
%     .max_particles - Počet částic inicializovat
%   public_vars    - Výstupní struktura
%
% VÝSTUP:
%   public_vars.particles - N×3 matice: [x, y, θ]
%   public_vars.weights   - N×1 vektor jednotných vah = 1/N
%
% VÝPOČETNÍ SLOŽITOST:
%   - Worst case: O(N * M) kde N = částice, M = iterace rejection sampling
%   - Zavisi na poměru volného prostoru vůči celkové mapě
%   - Malé mapy s velkými překážkami: pomaleji (více zamítnutí)
%   - Otevřené mapy: rychleji (více akceptací)
%
% ALTERNATIVNÍ PŘÍSTUPY (neimplementováno):
%   1. Importance sampling: inicializovat z zašuměného odhadu
%   2. Grid-based: pokrýt mapu mřížkou částic
%   3. Gaussian: inicializovat kolem odhadované pozice

% Extrahuj hranice mapy
x_min = read_only_vars.map.limits(1,1);
x_max = read_only_vars.map.limits(1,3);
y_min = read_only_vars.map.limits(1,2);
y_max = read_only_vars.map.limits(1,4);

% Načti seznam stěn (pro ověření volného prostoru)
walls = read_only_vars.map.walls;

% Počet částic k inicializaci
num_particles = read_only_vars.max_particles;

% Inicializuj pole na uložení částic
particles = zeros(num_particles, 3);

% Funkce pro ověření, zda je pozice ve volném prostoru
is_valid_position = @(x, y) is_inside_free_space(x, y, walls);

% -------- KROKU 1: Rejection Sampling - Generuj N náhodných částic ve volném prostoru --------
count = 0;  % Počet úspěšně inicializovaných částic

while count < num_particles
    % STEP 1: Náhodná pozice v obdélníku mapy
    x_rand = x_min + (x_max - x_min) * rand();   % random x
    y_rand = y_min + (y_max - y_min) * rand();   % random y
    
    % STEP 2: Náhodný úhel orientace v [-π, π]
    theta_rand = -pi + (2 * pi) * rand();
    
    % STEP 3: Ověř, zda je pozice ve volném prostoru
    if is_valid_position(x_rand, y_rand)
        % Pozice je PLATNÁ - přidej do pole
        count = count + 1;
        particles(count, :) = [x_rand, y_rand, theta_rand];
    end
    % Pokud pozice NE je platná: zahoď a zkus znovu (while loop pokračuje)
end

% -------- KROKU 2: Inicializuj jednotné váhy --------
% Na začátku: všechny částice mají stejnou pravděpodobnost (nedošlo k měření)
public_vars.particles = particles;                      % N×3: pozice částic
public_vars.weights = ones(num_particles, 1) / num_particles;  % N×1: jednotné váhy (1/N každá)

% POZNÁMKA: Váhy budou REWEIGHTED při prvním měření v weight_particles.m
%           Zde jsou jednotné jen proto, že nemáme žádnou informaci.

end
////////////////////////////////////////////////////////////////////////////////////////////////
function inside = is_inside_free_space(x, y, walls)
%IS_INSIDE_FREE_SPACE Detekce kolize - ověr, zda je pozice ve volném prostoru
%
% ÚČEL:
%   Určí, zda daná pozice (x, y) je BEZPEČNĚ V MAPĚ
%   (ne v nebo příliš blízko překážky/stěny).
%   Používá se pro validaci částic během inicialisace
%   a pro detekci kolizí v plánu pohybu.
%
% ALGORITMUS (Vzdálenost bodů od linií):
%   Pro KAŽDOU STĚNU v mapě:
%   1. Stěna je definována jako úsečka: [x1, y1] až [x2, y2]
%   2. Vypočítej KOLMOU VZDÁLENOST od bodu (x, y) k linii stěny
%   3. Použij vzorec: d = |ax + by + c| / √(a² + b²)
%      kde linie je: (y2-y1)*X - (x2-x1)*Y + x2*y1 - y2*x1 = 0
%   4. Pokud vzdálenost < práh (0.1 m): uvnitř stěny ❌ vrať FALSE
%   5. Pokud projdeme všechny stěny OK: vrať TRUE
%
% KOLMÁ VZDÁLENOST VYSVĚTLENÍ:
%        Stěna
%      (x1,y1) _________(x2,y2)
%              |\        
%           d  | \      ← KOLMÁ vzdálenost
%              |  \     
%              •(x,y)   ← testovaný bod
%
%   d = nejkratší vzdálenost od bodu k přímce stěny
%   Pokud d < 0.1 m: bod je příliš blízko stěny (nebezpečný)
%
% PARAMETR BEZPEČNOSTI:
%   práh = 0.1 m (10 cm)
%   - Zajišťuje, že robot/particle má vždycky alespoň 10 cm mezery od stěny
%   - Kompromis: dostatečná bezpečnost bez příliš malého volného prostoru
%   - Měl by být ≥ poloměr robota
%
% MATEMATICKÁ FORMULACE:
%   Bod P = (x, y)
%   Linie AB: A = (x1, y1), B = (x2, y2)
%   
%   Normalizovaná vzdálenost:
%   d = |AP × AB| / |AB|
%     = |(y2-y1)·x - (x2-x1)·y + x2·y1 - y2·x1| / √((y2-y1)² + (x2-x1)²)
%
% REFERENCE:
%   - Geometrie: point-to-line distance
%   - Používá se v INICIALIZACI a PLÁNOVÁNÍ POHYBU
%
% VSTUP:
%   x, y  - Testovaná pozice
%   walls - N×4 matice: [x1, y1, x2, y2]
%           Každý řádek = jedna stěna (úsečka)
%
% VÝSTUP:
%   inside - BOOLEAN:
%       TRUE = pozice je MIMO všechny stěny (bezpečná)
%       FALSE = pozice je V NEBO BLÍZKO stěny (nebezpečná)
%
% PŘÍKLAD:
%   inside = is_inside_free_space(5.0, 3.0, walls)
%   → TRUE: pozice [5, 3] je v pořádku
%   
%   inside = is_inside_free_space(0.05, 0.0, walls)
%   → FALSE: pozice [0.05, 0] je příliš blízko stěny na x=0
%
% AUTHOR POZNÁMKA:
%   Algoritmus předpokládá, že stěny jsou úsečky.
%   Vrací vzdálenost od LINKY (ne od úsečky).
%   V praxi to funguje dobře, pokud je práh dostatečně malý.

% Inicializuj status: předpokládej, že pozice JE ve volném prostoru
inside = true;

% -------- Kontrola KAŽDÉ STĚNY v mapě --------
for i = 1:size(walls, 1)
    % Extrahuj koncové body stěny
    x1 = walls(i, 1);  % počátek stěny
    y1 = walls(i, 2);
    x2 = walls(i, 3);  % konec stěny
    y2 = walls(i, 4);
    
    % -------- Vypočítej KOLMOU VZDÁLENOST od bodu k linii stěny --------
    % Vzorec: d = |ax + by + c| / √(a² + b²)
    % kde linie: (y2-y1)*x - (x2-x1)*y + x2*y1 - y2*x1 = 0
    
    numerator = abs((y2 - y1) * x - (x2 - x1) * y + x2 * y1 - y2 * x1);
    denominator = sqrt((y2 - y1)^2 + (x2 - x1)^2);
    
    d = numerator / denominator;  % KOLMÁ VZDÁLENOST [metry]
    
    % -------- Ověř, zda je pozice TOO CLOSE k této stěně --------
    if d < 0.02  % Bezpečnostní práh = 2 cm (sníженý z 10 cm pro lepší rovnoměrnost)
        inside = false;  % ❌ Pozice je příliš blízko stěny!
        return;          % Ukončení - žádná potřeba kontrolovat další stěny
    end
end

% Pokud jsme zde: všechny stěny OK, pozice je bezpečná ✓

end

function [particles] = update_particle_filter(read_only_vars, public_vars)
%UPDATE_PARTICLE_FILTER Hlavní smyčka částicového filtru
%
% ÚČEL:
%   Orchestruje tři klíčové kroky částicového filtru v každé iteraci:
%   1. PREDIKCE (PREDICTION): Předpoviď, kde budou částice po pohybu
%   2. KOREKCE (CORRECTION): Porovnej s měřením, vyvaž částice
%   3. PŘEVZORKOVÁNÍ (RESAMPLING): Odstraň slabé částice, duplikuj silné
%
% TEORETICKÝ ZÁKLADY:
%   Částicový filtr = SEKVENCIÁLNÍ MONTE CARLO
%   - Reprezentuje věrohodnost (belief) o poloze jako soubor částic
%   - Každá částice = jedna hypotéza "robot je na pozici [x, y, θ]"
%   - Váha = jak moc je tato hypotéza pravděpodobná
%
%   Cyklus:
%        Iterace n                    Iterace n+1
%   ┌──────────────┐              ┌──────────────┐
%   │  Částice 1   │ PREDICT      │  Částice 1'  │ 
%   │    (x,y,θ)   │─────────────►│  (x',y',θ')  │
%   │    w=0.1     │              │    w=0.1     │ ← Váha ještě stará
%   └──────────────┘              └──────────────┘
%        ...                           ...
%        Iterace n              MEASURE + WEIGHT
%   ┌──────────────┐              ┌──────────────┐
%   │  Částice N   │              │  Částice N'  │
%   │    (x,y,θ)   │ ────LIDAR──► │  (x',y',θ')  │
%   │    w=0.05    │              │    w=0.001   │ ← Nová váha!
%   └──────────────┘              └──────────────┘
%                                   RESAMPLE
%                              ┌──────────────┐
%                              │  Částice 1'' │ ← Často duplikát tě s w=0.1
%                              │   (s šumem)   │
%                              │    w=1/N     │ ← Unit. váhy po resamp.
%                              └──────────────┘
%
% VSTUP:
%   read_only_vars - Struktura s konfigurací:
%     .map              - Mapa stěn (pro měření)
%     .lidar_config     - Úhly paprsků LIDAR
%     .lidar_distances  - Naměřené vzdálenosti z robota
%     .sampling_period  - dt [sekundy]
%   public_vars    - Aktuální stav:
%     .particles        - N×3 matice pozic [x, y, θ]
%     .motion_vector    - [v_right, v_left] z plánu pohybu
%
% VÝSTUP:
%   particles    - N×3 matice aktualizovaných (po 3 krocích) částic
%
% VÝPOČETNÍ SLOŽITOST:
%   O(N*M) kde N = počet částic, M = počet LIDAR paprsků
%   Typicky 500 částic × 24 paprsků = 12,000 operací/iteraci
%   Doba execuce: ~50-100 ms na moderním procesoru
%
% DESIGN POZNÁMKA:
%   Tři kroky se vykonávají SEKVENČNĚ (ne paralelně):
%   1. Predict: Všechny částice aktualizuj pohybem
%   2. Correct: Všechny částice změř a vyvaž
%   3. Resample: Vyberi částice podle vah
%   
%   Pořadí je důležité! (musí být přesně PREDICT → CORRECT → RESAMPLE)
%   - predict_pose: N iterací (po částicích)
%   - compute_lidar_measurement: N iterací (po částicích)  
%   - weight_particles: O(N*M) operace (N×M matice)
%   - resample_particles: O(N log N) (cumsum)
%
% REFERENCE:
%   - PDF str. 10-12: 1D příklad sekvence PREDICT → MEASURE → RESAMPLE
%   - Probabilistic Robotics (Thrun et al.): Particle Filter Algorithm 4.4
%
% POKRAČOVÁNÍ - Viz příslušné funkce:
%   - predict_pose.m: Detaily diferenciálního pohonu + šumu
%   - compute_lidar_measurement.m: Ray-casting model měření
%   - weight_particles.m: Gaussova věrohodnost
%   - resample_particles.m: Systematické převzorkování

% ========== KROK 1: PREDIKCE (Predict Step) ==========
% Pro KAŽDOU ČÁSTICI: aplikuj pohybový povel a přidej šum

particles = public_vars.particles;  % Načti aktuální částice

for i=1:size(particles, 1)
    % Předpověz novou pozici na základě:
    % - Starší pozice (particles(i,:))
    % - Pohybového povelu (motion_vector: např. [0.5, 0.3] m/s)
    % - Šumu modelujícího reálnou neurčitost
    particles(i,:) = predict_pose(particles(i,:), public_vars.motion_vector, read_only_vars);
end

% PO KROKU 1 (PREDICT):
%   - Částice se posunula dle pohybu
%   - Šum rozšířil rozptyl (částice se "rozprostřely" víc)
%   - LIDAR měření by MĚLO být z NOVÉ předpověděné polohy

% ========== KROK 2: KOREKCE (Correction Step) ==========
% Pro KAŽDOU ČÁSTICI: spočítej co by měla "vidět" na LIDAR
% Porovnej s tím, co robot SKUTEČNĚ vidí
% Vypočítej váhu (jak moc věříš, že částice je na správném místě)

% STEP 2a: Vypočítej PŘEDPOKLÁDANÁ měření pro všechny částice
measurements = zeros(size(particles,1), length(read_only_vars.lidar_config));

for i=1:size(particles, 1)
    % Pro i-tou částici na pozici [x_i, y_i, θ_i]:
    % Co by měl LIDAR "vidět"? (ideální měření bez šumu)
    measurements(i,:) = compute_lidar_measurement(read_only_vars.map, particles(i,:), read_only_vars.lidar_config);
end
% measurements = N × M matice
%   N = počet částic
%   M = počet LIDAR paprsků
%   Element (i,j) = vzdálenost, kterou by j-tý paprsek měřil z i-té částice

% STEP 2b: Porovnej PŘEDPOKLÁDANÁ měření s AKTUÁLNÍMI měřeními z robota
% Vypočítej váhu: particles s měřením bližším skutečnosti dostanou vyšší váhu
weights = weight_particles(measurements, read_only_vars.lidar_distances);
% weights = N × 1 vektor
%   Element i = pravděpodobnost (váha) que částice i je správná
%   Součet(weights) = 1 (normalizace)

% PO KROKU 2 (CORRECT):
%   - Částice s "dobrými" měřeními mají vysokou váhu
%   - Částice s "špatnými" měřeními mají nízkou váhu
%   - Ale počet částic je stejný (jen se zmenily váhy)

% ========== KROK 3: PŘEVZORKOVÁNÍ (Resampling Step) ==========
% Problém degenerace: v průběhu času jen pár částic má vysokou váhu,
% zbytek má váhu téměř nula. Filtr "kolabuje".
% Řešení: Reduplmulate parts ze starých populované podle vah.
% Nízko-váhové částicí se SMAŽÍ, vysoke-vahovéí se DUPLIKUJÍ.

particles = resample_particles(particles, weights);

% PO KROKU 3 (RESAMPLE):
%   - Počet částic stále = N
%   - Ale nyní jsou rozprostřeny podle měřením (součást "konverguje")
%   - Všechny nové váhy = 1/N (jednotné, protože jsme právě převzorkovali)
%   - Přidán šum = částice nejsou ÚPLNĚ identické (zachovává se explorace)
%   - Dalším cyklus: PREDICT bude mít tuto novou populaci
%
% ========== CYKLUS KOMPLETNÍ ==========
% Výstup: particles = nová populace po 3 krocích
% Připravená pro další iteraci: PREDICT → CORRECT → RESAMPLE → ...

end

function [new_pose] = predict_pose(old_pose, motion_vector, read_only_vars)
%PREDICT_POSE Predikce pohybu robota pomocí diferenciálního modelu
%
% ÚČEL:
%   Implementuje MOTION MODEL (krok PREDIKCE) částicového filtru.
%   Aktualizuje polohu každé částice podle pohybového povelu,
%   a přidává šum modelující:
%   - Neurčitost řízení motorů (chyby tahu)
%   - Skluz kolék a zmeny povrchu (tření)
%   - Drift gyroskopu a chyby čidla (záhlaví)
%
% ALGORITMUS (Diferenciální pohon - Differential Drive):
%   1. Extrahuj aktuální polohu (x, y, θ) a měrné vstupy (v_r, v_l)
%   2. Konvertuj rychlosti kol na lineární a únglovou rychlost:
%      v = (v_r + v_l) / 2         [lineární rychlost, m/s]
%      ω = (v_r - v_l) / L         [únglова rychlost, rad/s]
%      kde L = vzdálenost mezi koly
%   
%   3. Aplikuj STOCHASTICKÝ ŠUM na rychlosti (modeluje reálnou neurčitost):
%      σ_v = K_v * |v|             [šum rychlosti roste s rychlostí]
%      σ_ω = K_ω * |ω|             [šum rotace roste s rotácí]
%      σ_θ = konstanta              [šum záhlaví nezávislý na pohybu]
%      
%   4. Vypočítej novou polohu pomocí diferenciálního KINEMATICS:
%      Pokud |ω| > prah:  vzorec pro "zakrivený pohyb" (únhlova rychl ≠ 0)
%         Δθ = ω * dt
%         Δx = (v/ω) * [sin(θ+Δθ) - sin(θ)]
%         Δy = (v/ω) * [cos(θ) - cos(θ+Δθ)]
%      Jinak: vzorec pro "přímochár" (aby se vyhloš dělení nulou)
%         Δx = v * cos(θ)
%         Δy = v * sin(θ)
%      θ_nowy = θ + ω*dt + šum, normalizováno na [-π, π]
%
% KLÍČOVÉ PARAMETRY (zlačované pro konvergenci):
%   K_v = 0.04 (4% lineární rychlosti): relativní šum rychlosti
%         Vyššího → více rozšířených částic, pomalejší konvergence ale robustnější
%         Nížšího → těsnější predikce, rychlejší konvergence ale možné chíby
%   K_ω = 0.04 (4% únglové rychlosti): relativní šum rotace
%   σ_θ = 0.04 rad: absolutní šum záhlaví (~2,3 stupně)
%
% REFERENCE:
%   - PDF str. 13: Případ "fanšapový" rozptýl částic po pohybu (vliv šumu)
%   - "Probabilistic Robotics" (Thrun et al.): Casti diferenciálního pohonu
%
% VSTUP:
%   old_pose      - Aktuální poloha: [x, y, θ] v metrech a radiánech
%   motion_vector - Rychlosti kol: [v_pravé, v_levé] v m/s
%   read_only_vars - Konfigurační struktura:
%     .sampling_period      - Časový krok dt [sekundy]
%     .agent_drive.interwheel_dist - L [metry]
%
% VÝSTUP:
%   new_pose      - Aktualizovaná poloha: [x_nová, y_nová, θ_nové]
%
% POZNÁMKA AUTORA:
%   Toto je MOTION MODEL (krok predikce). Šum modeluje reálnou neurčitost.
%   Bez šumu by všechny částice se stejnými povely následovaly identickou cestu
%   a filtr by kolaboval. Šum šíří částice (zvětšuje rozptyl),
%   důležité pro zachování nejakostitosti částic.

% Extrahuj souřádnice aktuální polohy
x = old_pose(1);
y = old_pose(2);
theta = old_pose(3);

% Extrahuj parametry
dt = read_only_vars.sampling_period;
v_l = motion_vector(2);  % rychlost levého kola
v_r = motion_vector(1);  % rychlost pravého kola
L = read_only_vars.agent_drive.interwheel_dist;  % vzdálenost mezi koly

% -------- KROK 1: Konvertuj rychlosti kol na lineární a únglovou rychlost --------
v = (v_r + v_l) / 2;      % lineární rychlost [m/s]
w = (v_r - v_l) / L;      % únglova rychlost [rad/s] OPRAVENO: (v_r - v_l) aby se shodoval s inverzní kinematickou

% -------- KROK 2: Aplikuj STOCHASTICKÝ ŠUM (modeluje reálnou neurčitost) --------
% Šum roste s velikostí pohybu (realistické: ryčhlejsmeji pohyb = větví chyba)
sigma_v = 0.04 * abs(v);      % šum rychlosti: 4% lineární rychlosti
sigma_w = 0.04 * abs(w);      % šum rotace: 4% rotáční frekvence
sigma_theta = 0.04;           % šum záhlaví: ~2,3 stupně (konstanta)

% Vzörkuj šum z Gaussova rozdelení (randn = N(0,1))
v_noisy = v + sigma_v * randn;      % šumčená lineární rychlost
w_noisy = w + sigma_w * randn;      % šumčená únglova rychlost

% -------- KROK 3: Aplikuj DIFERENCIALNI KINEMATIKU se šumčenými rychlostmi --------
% Ověř, singularitu: pokud únglova rychlost ≈ 0, použij aproximaci přímky
if abs(w_noisy) > 1e-3      % prah = 0.001 rad/s (přímka vs krivka)
    % ZAKERŠENÝ POHYB: nenulova ánglova rychlost
    % Exaktní vzorec pro diferenciální pohon
    delta_theta = w_noisy * dt;      % úhel rotace (OPRAVENO: bez negace)
    x_new = x + (v_noisy / w_noisy) * (sin(theta + delta_theta) - sin(theta));
    y_new = y + (v_noisy / w_noisy) * (cos(theta) - cos(theta + delta_theta));
else
    % PŘÍMOCHÁRÝ POHYB: únglova rychlost ≈ 0
    % Vyhneme se dělení nulou, jednodueše jedéme přímo
    x_new = x + v_noisy * cos(theta);
    y_new = y + v_noisy * sin(theta);
end

% Aktualizuj úhel orienstace a přidej независímý šum záhlaví
theta_new = theta + w_noisy * dt + sigma_theta * randn;  % OPRAVENO: bez negace

% Normalizuj úhel do rozmezí [-π, π] (zajiští konzistenci)
theta_new = mod(theta_new + pi, 2 * pi) - pi;

% Vrať aktualizovanou polohu
new_pose = [x_new, y_new, theta_new];

end


////////////////////////////////////////////////////////////////////////////////////
function [measurement] = compute_lidar_measurement(map, pose, lidar_config)
%COMPUTE_LIDAR_MEASUREMENT Simuluje LIDAR měření z pozice částice
%
% ÚČEL:
%   Implementuje MODEL MĚŘENÍ (korekční krok) částicového filtru.
%   Pro danou polohu částice predikuje, jaká LIDAR měření BY MĚLA být
%   pokud by byla částice na správné pozici robota. Používá se k porovnání
%   se skutečným měřeními senzoru pro vážení částic.
%
% ALGORITMUS (Ray-casting model měření):
%   1. Pro každý paprsek LIDAR čidla:
%      - Vypočítej směr paprsku v světové souřadné soustavě: θ_paprsek = θ_robota + offset_paprsku
%      - Vyměť paprsek (ray) z pozice robota ve směru paprsku
%      - Najdi průsečíky paprsku se všemi stěnami na mapě
%      - Vyber NEJBLIŽŠÍ průsečík (první povrch, který paprsek narazí)
%      - Změř vzdálenost k průsečíku
%   2. Vrátí vektor vzdáleností pro všechny paprsky
%
% KLÍČOVÝ KONCEPT:
%   LIDAR měří vzdálenosti k překážkám. Tato funkce počítá,
%   jaké vzdálenosti by měla "vidět" částice na své předpokládané pozici.
%   Porovnání s AKTUÁLNÍM měřením říká částicím, jaké mají být váhy:
%   - Pokud predikce ≈ skutečnost → Vysoká váha částice (pravděpodobně správná)
%   - Pokud predikce ≠ skutečnost → Nízká váha částice (pravděpodobně špatná)
%
% REFERENCE:
%   - PDF strana 14: Ukazuje koncept modelu měření s Gaussovy věrohodností
%   - Utility ray_cast(): trasuje paprsky skrz strukturu stěn
%
% VSTUP:
%   map        - Struktura mapy se stěnami:
%     .walls   - N×4 matice kde každý řádek = [x1, y1, x2, y2] (úsek stěny)
%   pose       - Poloha částice: [x, y, θ]
%   lidar_config - Vektor úhlů paprsků LIDAR relativně k robotovi: [φ1, φ2, ...]
%                  Délka určuje počet paprsků
%
% VÝSTUP:
%   measurement - Vektor vzdáleností (délka = počet_paprsků)
%                 Každý prvek = vzdálenost k nejbližšímu průsečíku se stěnou [metry]
%                 Hodnota = inf pokud se paprsek neprotne se žádnou stěnou
%
% POZNÁMKA:
%   Vrací IDEÁLNÍ (bezšumová) měření. Reálný šum senzoru se přidá
%   později v weight_particles() při porovnání predikce vs skutečné měření.

% Extrahuj souřadnice polohy částice
x = pose(1);
y = pose(2);
theta = pose(3);

% Zjisti počet paprsků LIDAR
num_beams = length(lidar_config);
measurement = zeros(1, num_beams);

% -------- Pro každý paprsek LIDAR vypočítej vzdálenost k první ploše --------
for i = 1:num_beams
    % Vypočítej směr paprsku v globální souřadné soustavě
    % (přidej offset paprsku k orientaci robota)
    beam_angle = theta + lidar_config(i);
    
    % KROK 1: Vyměť paprsek z pozice částice ve směru paprsku
    %         Vrátí všechny průsečíky s stěnami (pokud existují)
    intersections = ray_cast([x; y], map.walls, beam_angle);
    
    % KROK 2: Najdi nejbližší průsečík (první plocha, kterou paprsek narazí)
    if ~isempty(intersections)
        % Vypočítej vzdálenosti od částice ke každému průsečíku
        dx = intersections(:, 1) - x;  % Rozdíl x-souřadnic
        dy = intersections(:, 2) - y;  % Rozdíl y-souřadnic
        dist = sqrt(dx.^2 + dy.^2);    % Euklidovská vzdálenost
        
        % Vyber minimální vzdálenost (nejbližší průsečík se stěnou)
        measurement(i) = min(dist);
    else
        % Žádný průsečík - paprsek "uletěl" mimo mapu
        % Použij nekonečno aby se signalizovalo "bez detekce"
        measurement(i) = inf;
    end
end

% POZNÁMKA: Tato funkce vrací IDEÁLNÍ měření (bez šumu).
%           Skutečný LIDAR senzor má šum, ale ten se řeší v weight_particles()
%           při výpočtu podobnosti mezi predikovaným a měřeným měřením.

end

function [new_particles] = resample_particles(particles, weights)
%RESAMPLE_PARTICLES Převzorkování k odstranění degenerovaných částic
%
% ÚČEL:
%   Řeší PROBLÉM DEGENERACE ČÁSTIC: v průběhu času mají jen pár částic
%   významnou váhu, zbytek se stává zanedbatelný. To způsobuje:
%   - Ztrátu rozmanitosti → filtr se nemůže vzpamatovat z lokalizačních chyb
%   - Ztrátu přesnosti → pár částic nemůže reprezentovat plné přesvědčení
%   
%   Převzorkování vytváří novou populaci výběrem vysokováhových částic
%   vícekrát, a odstraňuje částice nízkováhové. Počet částic zůstává
%   stejný, ale s lepší distribucí.
%
% ALGORITMUS (Systematické převzorkování - nízkovarianční metoda):
%   1. Vypočítej kumulativní distribuci vah (CDF)
%   2. Vygeneruj N rovnoměrně vzdálených vzorků s jedním náhodným offsetem:
%      u_i = (r + i-1) / N kde r = náhodné číslo v [0, 1/N]
%   3. Pro každý vzorek u_i najdi kterou částici reprezentuje pomocí CDF
%      (částice j kde CDF[j-1] < u_i ≤ CDF[j])
%   4. Vyber tu částici pro novou generaci (možné duplikování)
%   5. Přidej šum duplikátům (zabrání degeneraci do jednoho bodu)
%
% POROVNÁNÍ S ALTERNATIVAMI:
%   - Ruletové kolo: částice se vybírají jako otáčení ruletky
%                    náhodné offsety, vyšší variace
%   - Systematické (tato metoda): deterministické rozestupy optimálně rozprostřou
%                                 nižší variace, stabilnější
%   - Reziduální: deterministické + náhodná komponenta
%   
%   Systematické převzorkování je nejlepší: nízká variace + efektivní + stabilní
%
% ANALOGIE S RULETOVÝM KOLEM (z PDF strana 15-18):
%   Představ si váhy jako kusy na kole:
%   - Vysoká váha → velký kus
%   - Nízká váha → malý kus
%   Otáčej kolo N-krát: pokaždé, když se jehla zastaví, vyber tu částici.
%   Systematické metoda: otáčíš jednou s N rovnoměrně rozloženými značkami.
%
% INJEKCE ŠUMU (Kritické pro částicový filtr):
%   Po výběru částic se přidá malý Gaussovský šum aby se zabránilo:
%   - Zubožení vzorků: všechny částice se po převzorkování stanou stejné
%   - Ztrátě reprezentace nejistoty
%   - Kolapsu filtru když všechny částice ukazují na jedno místo
%   
%   Parametry šumu:
%   σ_x = 0.1 m: šum x-pozice (10 cm)
%   σ_y = 0.1 m: šum y-pozice (10 cm)  
%   σ_θ = 0.05 rad: šum orientace (~3 stupně)
%
%   TUNOVÁNÍ:
%   - Málo šumu: částice se sesypou na jedno místo (divergence)
%   - Moc šumu: částice se rozprostřou, těžko konvergují
%   - Default [0.1, 0.1, 0.05] balancuje průzkum vs. vykořisťování
%
% REFERENCE:
%   - PDF strana 15: "Ruletové kolo" - vizuální vysvětlení
%   - PDF strana 18: Shluku částic po převzorkování (efekt šumu)
%   - Probabilistic Robotics (Thrun et al.): Sekce o systematickém převzorkování
%
% VSTUP:
%   particles - N×3 matice poloh částic [x, y, θ]
%   weights   - N×1 vektor vah částic (důležitostní hodnoty)
%              Měl by součet k 1 (zde bude normalizován)
%
% VÝSTUP:
%   new_particles - N×3 převzorkované částice s přidaným šumem
%                   Stejný počet, redistribuované podle důležitosti
%
% MATEMATICKÝ VZOREC:
%   Pro každé i=1..N:
%     u_i = (r + i-1) / N  kde r ~ Uniform(0, 1/N)
%     Vyber částici j kde: CDF[j-1] < u_i ≤ CDF[j]
%     x'_i = x_j + šum_x ~ N(0, σ_x²)
%     y'_i = y_j + šum_y ~ N(0, σ_y²)
%     θ'_i = wrapTo2Pi(θ_j + šum_θ ~ N(0, σ_θ²))
%
% POZNÁMKA:
%   Toto je KROK PŘEVZORKOVÁNÍ částicového filtru.
%   Nezbytné pro boj proti degeneraci - zajistí zachování rozmanitosti.
%   Šum zabraňuje kolapsu zatímco systematická metoda zajistí stabilitu.
%   Výpočetní čas: O(N log N) pro třídění/CDF operace

% Zjisti počet částic
N = size(particles, 1);

% PARAMETRY INJEKCE ŠUMU (ladicí)
% Zabraňují zubožení vzorků při zachování reprezentace nejistoty
sigma = [0.1, 0.1, 0.05];  % [σ_x, σ_y, σ_θ] v metrech a radiánech

% Inicializuj pole nových částic
new_particles = zeros(N, 3);

% Normalizuj váhy (zajisti součet = 1)
weights = weights / sum(weights);

% -------- KROK 1: Vypočítej kumulativní distribuci vah (CDF) --------
% Použije se k mapování náhodných vzorků na částice
% Pokud váhy = [0.1, 0.5, 0.3, 0.1]
% Pak CDF  =  [0.1, 0.6, 0.9, 1.0]
cumulative_weights = cumsum(weights);

% -------- KROK 2: Vygeneruj POZICE SYSTEMATICKÝCH VZORKŮ --------
% Náhodný offset aby se nezabraňovalo vybírání stejných částic
r = rand / N;  % Náhodné číslo v [0, 1/N]

% Index pro hledání v CDF (ukazatel)
j = 1;

% -------- KROK 3: Pro každou novou částici najdi odpovídající starou --------
for i = 1:N
    % SYSTEMATICKÝ VZOREK: rovnoměrně vzdálené s náhodným offsetem
    %   Vytváří N vzorků na pozicích: r, r+1/N, r+2/N, ..., r+(N-1)/N
    %   Tato rozestupy zajistí jednotné pokrytí zatímco náhodné r přidá variabilitu
    U = r + (i - 1) / N;
    
    % BINÁRNÍ HLEDÁNÍ: Najdi částici j kde CDF[j-1] < U ≤ CDF[j]
    % Tato částice bude vybrána (možně vícekrát)
    while U > cumulative_weights(j)
        j = j + 1;
    end
    
    % Zkopíruj vybranou částici do nové generace
    new_particles(i, :) = particles(j, :);
end

% -------- KROK 4: INJEKCE ŠUMU --------
% Přidej Gaussovský šum aby se zabránilo všem částicím být identické
% To zachovává nejistotu a umožňuje úniku z lokálních minim
noise = randn(N, 3) .* sigma;  % N×3 náhodná matice škálovaná σ
new_particles = new_particles + noise;

% Sviň úhly orientace zpět do rozmezí [-π, π] (normalizuj úhly)
new_particles(:, 3) = wrapToPi(new_particles(:, 3));

% Po převzorkování:
%   - Degenerované (nízkovážné) částice jsou odstraněny
%   - Dobré (vysokováhové) částice jsou duplikovány
%   - Rozmanitost zachována injektováním šumu
%   - Počet částic nezmění
%   - Připraveny pro další krok predikce

end
